#include <iostream>
#include <vector>
#include <ctime>

using namespace std;

// Heapify function to maintain heap property
void heapify(vector<int>& arr, int n, int i) {
    int largest = i;
    int left = 2 * i + 1;
    int right = 2 * i + 2;

    if (left < n && arr[left] > arr[largest]) {
        largest = left;
    }

    if (right < n && arr[right] > arr[largest]) {
        largest = right;
    }

    if (largest != i) {
        swap(arr[i], arr[largest]);
        heapify(arr, n, largest);
    }
}

// Heap sort function
void heapSort(vector<int>& arr) {
    int n = arr.size();

    // Build max heap
    for (int i = n / 2 - 1; i >= 0; i--) {
        heapify(arr, n, i);
    }

    // Heap sort
    for (int i = n - 1; i > 0; i--) {
        swap(arr[0], arr[i]);
        heapify(arr, i, 0);
    }
}

int main() {
    // Seed for random number generation
    srand(time(0));

    int n;
    cout << "Enter the number of elements: ";
    cin >> n;

    vector<int> numbers(n);

    // Generate random numbers
    for (int i = 0; i < n; i++) {
        numbers[i] = rand() % 1000;
    }

    cout << "Original list: ";
    for (int num : numbers) {
        cout << num << " ";
    }
    cout << endl;

    // Sort the numbers using Heap sort
    heapSort(numbers);

    cout << "Sorted list: ";
    for (int num : numbers) {
        cout << num << " ";
    }
    cout << endl;

    return 0;
}
